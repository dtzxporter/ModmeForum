function 24_zoms() //_zm.gsc 3717
{
start_at_round = 5;

while(1)
    {
    level waittill("zombie_total_set");

    if(level.round_number &lt; start_at_round) //skip if round nm is too low
        continue;

    if(GetPlayers().size == 1)
        level.zombie_total = 24;

    }

}
