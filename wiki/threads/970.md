{
    trig = GetEnt("shootable_trig", "targetname");
    model = GetEnt("shootable_model", "targetname");
 
    trig SetHintString("");
    trig SetCursorHint("HINT_NOICON");
 
    while(1)
    {
        trig_1 waittill("trigger", player)
 
        break;
    }
 
    trig Delete();
    model Delete();
  
}
