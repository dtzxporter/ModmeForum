function zombiekill()
{
	zombs = getaiarray( "axis" );
	level.zombie_total = 0;
	if( IsDefined( zombs ) )
	{
		i = 0;
		while( i &lt; zombs.size )
		{
			zombs[ i] dodamage( zombs[ i].health * 5000, ( 0, 0, 0 ), self );
			wait 0.05;
			i++;
		}
	}
}
